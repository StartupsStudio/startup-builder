### Decision Framework

```typescript
const portfolioDecisions = agenticFunction({
  name: "portfolioOptimization",
  schedule: "monthly",

  analyze: async () => {
    const startups = await db.startups.all()

    const analysis = startups.map(startup => ({
      startup,
      metrics: {
        mrr: getMRR(startup),
        growth: getGrowthRate(startup),
        retention: getRetentionRate(startup),
        cac: getCAC(startup),
        ltv: getLTV(startup)
      },
      stage: determineStage(startup),
      recommendation: generateRecommendation(startup)
    }))

    return {
      doubleDown: analysis.filter(s => s.recommendation === "invest-more"),
      maintain: analysis.filter(s => s.recommendation === "continue"),
      pivot: analysis.filter(s => s.recommendation === "change-approach"),
      sunset: analysis.filter(s => s.recommendation === "shutdown")
    }
  },

  actions: {
    doubleDown: async (startup) => {
      await increaseMarketing(startup, 2.0)
      await addFeatures(startup, startup.backlog.top5)
      await assignPM(startup)
    },

    maintain: async (startup) => {
      // Continue autonomous operations
    },

    pivot: async (startup) => {
      const pivotStrategy = await generatePivotStrategy(startup)
      await implement(pivotStrategy)
      await resetMetrics(startup)
    },

    sunset: async (startup) => {
      await notifyCustomers(startup, "migration-path")
      await archiveData(startup)
      await captureLearnings(startup)
      await shutdown(startup)
    }
  }
})
```
